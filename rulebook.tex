	This rulebook defines safe and fair use of the \Room\ for projects, storage, fabrication, and testing. It applies to all members, guests, and collaborators.

	\section*{A. Access \& Conduct}
	\begin{enumerate}
		\item \textbf{Access Hours.} Use the club only during posted access hours or with prior permission from allowed officials/permitted members in charge; biometric entry applies to approved \emph{active members} only.
		\item \textbf{Operational Holds.} The \textbf{club may temporarily stop access} for segregation of components, audits, cleaning, or maintenance. Such holds will be announced by email when possible.
		\item \textbf{Night Work (Recommendation).} Heavy/noisy or hazardous operations (drilling, grinding, extensive soldering) are \textbf{discouraged at night}. Aim to finish such operations \textbf{before 22:00}.
		\item \textbf{Sign-In/Out.} Record your name, roll number, \textbf{designation (Member / Active member / Guest / Official)}, in/out time, and purpose in the log book if available.
		\item \textbf{Guests.} Guests/visitors need prior permission, must be accompanied by a permitted member, and \textbf{may not use tools or equipment}.
		\item \textbf{Respect.} Treat people, tools, and ongoing projects with care. No harassment, vandalism, or disruptive behavior.
	\end{enumerate}
	
	% ============================================================
	\section*{B. Food, Drinks \& Cleanliness}
	\begin{enumerate}
		\item \textbf{Food.} Dry snacks are fine. Keep liquids far from electronics and machines.
		\item \textbf{Clean Up.} Clear waste immediately; wipe benches you used.
		\item \textbf{Spills.} If a spill occurs, stop work, disconnect affected gear (if safe), and email the club regarding the incident with full transparency.
	\end{enumerate}
	
	% ============================================================
	\section*{C. Storage \& Labelling}
	\begin{enumerate}
		\item \textbf{Project Bins on Tables Only.} Keep project bins/boxes \textbf{only on the tables}, not even on the standing desk at the center. Do not place project bins in the almira.
		\item \textbf{Almira Use (Sensitive/Expensive Projects).} Prior permission from allowed officials is required to use the almira for expensive/sensitive projects.
		\item \textbf{Labels.} Each bin must list project name and/or name of at least one team member.
		\item \textbf{Clearing.} Unattended/unclear items may be cleared after notice.
	\end{enumerate}
	
	% ============================================================
	\section*{D. Components: Taking, Returning, Reporting}
	\begin{enumerate}
		\item \textbf{Take Only What You Need.} Be conservative with passives and small parts.
		\item \textbf{Return To Correct Boxes.} If you used a resistor from the resistor box, put it back in the correct value drawer/strip. Same applies to transistors, diodes, ICs, headers, jumpers, dev-boards, sensors, motors, drivers, screws, nuts, washers. Keep assortments tidy.
		\item \textbf{If Unsure.} Don’t dump misc parts into random bins. Ask an allowed official or leave them in a clearly labelled “to-sort” tray, which will be available on the standing desk at the center.
		\item \textbf{Damage/Missing.} If a part/tool is damaged or missing, email \ClubEmail\ immediately. If you found it damaged, mention when it was last seen in good condition (date/time/person if known).
	\end{enumerate}

	\subsection*{E2. General Rules}
	\begin{enumerate}
		\item \textbf{Keep Hazards Away From 3D Printer.} Do not use drills, grinders, or soldering anywhere near the 3D printer.
		\item \textbf{Night Work (Recommendation).} Avoid drilling/grinding/extended soldering at night; target 22:00 as a soft cutoff for such operations.
		\item \textbf{Power Off.} After use, switch off instruments and unplug portable tools.
	\end{enumerate}
	
	
	% ============================================================
	\section*{F. 3D Printing}
	\begin{enumerate}
		\item \textbf{Channel.} All 3D print requests go by email to \ClubEmail.
		\item \textbf{Printer and materials.} Printer: \textbf{Bambu Lab A1}. Filament stocked: \textbf{PLA} (other filaments might be available).
		Maximum build volume: \textbf{25.6\,cm $\times$ 25.6\,cm $\times$ 25.6\,cm};
		recommended working envelope: \textbf{up to 24\,cm cube} for reliability.
		\item \textbf{Allowed use.} \textbf{Only scientific/academic project work}. Personal or non-scientific items (door hooks, generic holders, etc.) are \textbf{not allowed}.
		\item \textbf{Operation.} Printing is performed or supervised by allowed officials or permitted members. \textbf{No unsupervised operation.} Guests/visitors may not request or operate prints.
		\item \textbf{Models and files.} Prepare in \textbf{Bambu Studio}. \textbf{Sliced output is mandatory}; attach the \textbf{.gcode} file (required).
		\item \textbf{Safety near the printer.} \textbf{Do not drill, grind, or solder anywhere near the printer.} Keep the surrounding bench clear; do not touch the machine without permission.
	\end{enumerate}
	
	% ============================================================
	\section*{G. Equipment Use Duration \& Project Email (No Hard Limits)}
	\begin{enumerate}
		\item \textbf{No Fixed Time Limits.} There is \textbf{no formal time limit} on equipment use. Be considerate so others can proceed.
		\item \textbf{Mandatory Project Email.} For each project, send an email to \ClubEmail\ including:
		\begin{itemize}
			\item \textbf{Detailed list of all parts/components} you used (ICs, sensors, motors, passives, hardware, PCBs/boards, etc.).
			\item \textbf{Short project description} (goal, current status).
			\item \textbf{Assistance log:} which allowed officials/permitted members assisted you in using restricted tools (e.g., 3D printer, angle grinder), including date(s).
		\end{itemize}
	\end{enumerate}
	
	% ============================================================
	\section*{H. Electrical \& Test Safety}
	\begin{enumerate}
		\item \textbf{Know Your Limits.} If unsure about scope probes, PSU limits, or grounding—pause and ask a club official.
		\item \textbf{Voltage/Current Limits.} Set current limits before powering circuits. Double-check polarity and ratings.
		\item \textbf{Probes.} Use appropriate probes/ranges. Never exceed rated voltage.
	\end{enumerate}
	
	% ============================================================
	\section*{I. Housekeeping \& Lock-up}
	\begin{enumerate}
		\item \textbf{Before Leaving:}
		\begin{itemize}
			\item Power down instruments; unplug portable tools; close laptops.
			\item Return all tools/components to proper places; clear the bench; dispose of trash.
		\end{itemize}
		\item \textbf{Doors \& Windows.} Close and secure all the doors and windows, including terrace entry.
		\item \textbf{Lights \& Fans.} Switch off lights and fans.
	\end{enumerate}
